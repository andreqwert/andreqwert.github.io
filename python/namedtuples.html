Именованные кортежи


        from collections import namedtuple


        """
        ПОСТАНОВКА ЗАДАЧИ
        Пример:
        color = (55, 155, 255)
        print(color[0])    # 55
        Имеется такой код. Он не очень читабелен, поскольку спустя время забывается,
        что такое [0] или [1] и подобные индексы. Чтобы повысить читабельность кода в
        таких случаях - пользуются именованными кортежами.
        """

        Color = namedtuple('Color', ['red', 'green', 'blue'])

        color = Color(55, 155, 255)
        print(color.blue)    # 255
