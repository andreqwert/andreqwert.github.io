1. Линейный поиск (Linear search) 
    

    # A - массив
    # x - искомое значение

    def linear_search(A, x):
        """Linear search"""

        answer = 'NotFound'
        for i in range(0, len(A)):
            if A[i] == x:
                answer = i
        return answer


    linear_search(['A', 'B', 'C', 'D'], 'E') # NotFound
    linear_search(['A', 'B', 'C', 'D'], 'A') # 0
    linear_search(['A', 'B', 'C', 'D'], 'C') # 2
2. Рекурсия (Recursion) 
    
    def factorial(n):
        if n == 0:
            return 1
        else:
            return n*factorial(n-1)


    print(factorial(5)) # 120
